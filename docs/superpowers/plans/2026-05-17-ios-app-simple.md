**Goal:** Build a native-feeling iOS app (React Native + Expo) at `ios/` inside the existing repo, matching the web app's current design (dark navy, ambulance yellow, Fraunces serif, IBM Plex Mono), with 4 tabs (Manual, Códigos, Vademécum, Mapa) + search + full offline support.

```js
#!/usr/bin/env node
/**
 * Reads content from ../content/ and writes processed data to ios/src/data/
 * Run: node scripts/sync-content.js (from ios/ directory)
 */
const fs = require('fs')
const path = require('path')

const ROOT = path.resolve(__dirname, '../..')   // repo root
const CONTENT = path.join(ROOT, 'content')
const OUT = path.join(__dirname, '../src/data')

fs.mkdirSync(OUT, { recursive: true })

// ── Procedures ──────────────────────────────────────────────────────────────
const SECTION_COLORS = {
  sva: '#D71920',
  svb: '#0057B8',
  operativos: '#F97316',
  administrativos: '#64748B',
  comunicaciones: '#06B6D4',
  general: '#10B981',
  tecnicas: '#8B5CF6',
  psicologicos: '#DFFF00',
  drp: '#F97316',
  intervinientes: '#10B981',
}

function extractTitle(md) {
  const match = md.match(/^#\s+(.+)$/m)
  return match ? match[1].trim() : 'Sin título'
}

const procedures = []
const procDir = path.join(CONTENT, 'procedures')
const sections = fs.readdirSync(procDir).filter(d =>
  fs.statSync(path.join(procDir, d)).isDirectory()
)

for (const section of sections) {
  const sectionDir = path.join(procDir, section)
  const files = fs.readdirSync(sectionDir).filter(f => f.endsWith('.md'))
  for (const file of files) {
    const id = file.replace('.md', '')
    const content = fs.readFileSync(path.join(sectionDir, file), 'utf8')
    procedures.push({
      id,
      title: extractTitle(content),
      section,
      sectionColor: SECTION_COLORS[section] ?? '#64748B',
      content,
    })
  }
}

// Sort by ID numerically
procedures.sort((a, b) => {
  const na = parseInt(a.id, 10), nb = parseInt(b.id, 10)
  if (!isNaN(na) && !isNaN(nb)) return na - nb
  return a.id.localeCompare(b.id)
})

fs.writeFileSync(path.join(OUT, 'procedures.json'), JSON.stringify(procedures, null, 2))
console.log(`✓ ${procedures.length} procedures written to src/data/procedures.json`)

// ── Vademecum ────────────────────────────────────────────────────────────────
const vademecum = JSON.parse(fs.readFileSync(path.join(CONTENT, 'data/vademecum.json'), 'utf8'))
fs.writeFileSync(path.join(OUT, 'vademecum.json'), JSON.stringify(vademecum, null, 2))
console.log(`✓ ${vademecum.length} drugs written to src/data/vademecum.json`)

// ── Codigos (merge all codigos-*.json files) ──────────────────────────────────
const codigoFiles = fs.readdirSync(path.join(CONTENT, 'data'))
  .filter(f => f.startsWith('codigos-') && f.endsWith('.json'))

const codigosMap = {}
for (const file of codigoFiles) {
  const key = file.replace('codigos-', '').replace('.json', '')
  const data = JSON.parse(fs.readFileSync(path.join(CONTENT, 'data', file), 'utf8'))
  codigosMap[key] = data
}
fs.writeFileSync(path.join(OUT, 'codigos.json'), JSON.stringify(codigosMap, null, 2))
console.log(`✓ Codigos written (${codigoFiles.length} files) to src/data/codigos.json`)

// ── Hospitals ────────────────────────────────────────────────────────────────
const hospitals = JSON.parse(fs.readFileSync(path.join(CONTENT, 'data/hospitals.json'), 'utf8'))
fs.writeFileSync(path.join(OUT, 'hospitals.json'), JSON.stringify(hospitals, null, 2))
console.log(`✓ ${hospitals.length} hospitals written to src/data/hospitals.json`)

// ── Abreviaturas ─────────────────────────────────────────────────────────────
const abreviaturas = JSON.parse(fs.readFileSync(path.join(CONTENT, 'data/abreviaturas.json'), 'utf8'))
fs.writeFileSync(path.join(OUT, 'abreviaturas.json'), JSON.stringify(abreviaturas, null, 2))
console.log(`✓ ${abreviaturas.length} abreviaturas written to src/data/abreviaturas.json`)

console.log('\n✅ Content sync complete.')
```

```ts
import { useState, useMemo } from 'react'
import Fuse from 'fuse.js'

export function useSearch<T>(
  items: T[],
  keys: Array<keyof T & string>,
  options?: { threshold?: number }
) {
  const [query, setQuery] = useState('')

  const fuse = useMemo(
    () => new Fuse(items, { keys, threshold: options?.threshold ?? 0.3, includeScore: true, minMatchCharLength: 2 }),
    [items, keys, options?.threshold]
  )

  const results = useMemo(
    () => (query.length < 2 ? items : fuse.search(query).map(r => r.item)),
    [query, fuse, items]
  )

  return { query, setQuery, results }
}
```

- [ ] **Create useFavorites.ts**

```ts
import { useState, useEffect, useCallback } from 'react'
import AsyncStorage from '@react-native-async-storage/async-storage'
import * as Haptics from 'expo-haptics'

const KEY = '@samur/favorites'

export function useFavorites() {
  const [favorites, setFavorites] = useState<string[]>([])

  useEffect(() => {
    AsyncStorage.getItem(KEY).then(v => { if (v) setFavorites(JSON.parse(v)) })
  }, [])

  const toggle = useCallback(async (id: string) => {
    setFavorites(prev => {
      const next = prev.includes(id) ? prev.filter(f => f !== id) : [...prev, id]
      AsyncStorage.setItem(KEY, JSON.stringify(next))
      return next
    })
    Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Light)
  }, [])

  const isFavorite = useCallback((id: string) => favorites.includes(id), [favorites])

  return { favorites, toggle, isFavorite }
}
```

const MAX = 50

export function useHistory() {
  const [history, setHistory] = useState<string[]>([])

  useEffect(() => {
    AsyncStorage.getItem(KEY).then(v => { if (v) setHistory(JSON.parse(v)) })
  }, [])

  const push = useCallback(async (id: string) => {
    setHistory(prev => {
      const next = [id, ...prev.filter(h => h !== id)].slice(0, MAX)
      AsyncStorage.setItem(KEY, JSON.stringify(next))
      return next
    })
  }, [])

  return { history, push }
}
